Header <boost/text/text_view.hpp>

namespace boost {
  namespace text {
    template<typename CPIter> struct grapheme_range;
    struct text_view;
    bool operator==(text_view lhs, text_view rhs);
    bool operator!=(text_view lhs, text_view rhs);
    text_view::iterator begin(text_view tv);
    text_view::iterator end(text_view tv);
    text_view::iterator cbegin(text_view tv);
    text_view::iterator cend(text_view tv);
    text_view::reverse_iterator rbegin(text_view tv);
    text_view::reverse_iterator rend(text_view tv);
    text_view::reverse_iterator crbegin(text_view tv);
    text_view::reverse_iterator crend(text_view tv);
    int operator+(text_view const & t, char const * c_str);
  }
}
