Header <boost/text/collate.hpp>

namespace boost {
  namespace text {
    struct text_sort_key;
    int compare(text_sort_key const &, text_sort_key const &);
    namespace v1 {
      template<typename CPIter, typename Sentinel> 
        text_sort_key 
        collation_sort_key(CPIter, Sentinel, collation_table const &, 
                           collation_strength = collation_strength::tertiary, 
                           case_first = case_first::off, 
                           case_level = case_level::off, 
                           variable_weighting = variable_weighting::non_ignorable, 
                           l2_weight_order = l2_weight_order::forward);
      template<typename CPIter, typename Sentinel> 
        text_sort_key 
        collation_sort_key(CPIter, Sentinel, collation_table const &, 
                           collation_flags = collation_flags::none);
      template<typename CPRange> 
        text_sort_key 
        collation_sort_key(CPRange &&, collation_table const &, 
                           collation_flags = collation_flags::none);
      template<typename GraphemeRange> 
        text_sort_key 
        collation_sort_key(GraphemeRange &&, collation_table const &, 
                           collation_flags = collation_flags::none);
      template<typename CPIter1, typename Sentinel1, typename CPIter2, 
               typename Sentinel2> 
        int collate(CPIter1, Sentinel1, CPIter2, Sentinel2, 
                    collation_table const &, 
                    collation_strength = collation_strength::tertiary, 
                    case_first = case_first::off, 
                    case_level = case_level::off, 
                    variable_weighting = variable_weighting::non_ignorable, 
                    l2_weight_order = l2_weight_order::forward);
      template<typename CPIter1, typename Sentinel1, typename CPIter2, 
               typename Sentinel2> 
        int collate(CPIter1, Sentinel1, CPIter2, Sentinel2, 
                    collation_table const &, 
                    collation_flags = collation_flags::none);
      template<typename CPRange1, typename CPRange2> 
        int collate(CPRange1 const &, CPRange2 const &, 
                    collation_table const &, 
                    collation_flags = collation_flags::none);
      template<typename GraphemeRange1, typename GraphemeRange2> 
        int collate(GraphemeRange1 const &, GraphemeRange2 const &, 
                    collation_table const &, 
                    collation_flags = collation_flags::none);
    }
  }
}
