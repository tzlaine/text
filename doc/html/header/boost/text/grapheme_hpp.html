Header <boost/text/grapheme.hpp>

namespace boost {
  namespace text {
    struct grapheme;
    template<typename CPIter> struct grapheme_view;
    int storage_bytes(grapheme const &);
    template<typename CPIter> int storage_bytes(grapheme_view< CPIter >);
    template<typename CPIter1, typename CPIter2> 
      bool operator==(grapheme_view< CPIter1 >, grapheme_view< CPIter2 >);
    template<typename CPIter1, typename CPIter2> 
      bool operator!=(grapheme_view< CPIter1 >, grapheme_view< CPIter2 >);
    template<typename CPIter1, typename CPIter2> 
      bool operator==(cp_range< CPIter1 >, grapheme_view< CPIter2 >);
    template<typename CPIter1, typename CPIter2> 
      bool operator==(grapheme_view< CPIter1 >, cp_range< CPIter2 >);
    template<typename CPIter1, typename CPIter2> 
      bool operator!=(cp_range< CPIter1 >, cp_range< CPIter2 >);
    template<typename CPIter1, typename CPIter2> 
      bool operator!=(grapheme_view< CPIter1 >, cp_range< CPIter2 >);
    template<typename CPIter> 
      bool operator==(grapheme const &, grapheme_view< CPIter >);
    template<typename CPIter> 
      bool operator==(grapheme_view< CPIter >, grapheme const &);
    template<typename CPIter> 
      bool operator!=(grapheme const &, grapheme_view< CPIter >);
    template<typename CPIter> 
      bool operator!=(grapheme_view< CPIter >, grapheme const &);
  }
}
